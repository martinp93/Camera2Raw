076  public static CompositeImage luminance(CompositeImage img) {
077    float[][] luminanceValues = new float[img.getHeight()][img.getWidth()];
078    for (int i = 0; i < img.getHeight(); i++) {
079      for (int j = 0; j < img.getWidth(); j++) {
080        luminanceValues[i][j] = luminance(img.getPixel(j, i));
081      }
082    }
083    return new CompositeImage(relativeLuminanceToColorDomain(luminanceValues));
084  }

089    return luminance(color.getRGB());

092  public static float luminance(int color) {
093    Color RGB = new Color(color);
094    double red = RGB.getRed() / 255.0;
095    red = red < THRESHOLD ? red / 12.92 : Math.pow((red + 0.055) / 1.055, 2.4);
096    double green = RGB.getGreen() / 255.0;
097    green = green < THRESHOLD ? green / 12.92 : Math.pow((green + 0.055) / 1.055, 2.4);
098    double blue = RGB.getBlue() / 255.0;
099    blue = blue < THRESHOLD ? blue / 12.92 : Math.pow((blue + 0.055) / 1.055, 2.4);
100    return (float) ((WEIGHT_RED * red) + (WEIGHT_GREEN * green) + (WEIGHT_BLUE * blue));
101  }

103  // Mapping straight from 0 - 255 to 0.0 - 1.0, no gamma correction or other correction method
104  private static double[][] convertToLinear(int[][] imageValues) {
105    double[][] result = new double[imageValues.length][imageValues[0].length];
106    for (int i = 0; i < imageValues.length; i++) {
107      for (int j = 0; j < imageValues[i].length; j++) {
108        result[i][j] = (double)imageValues[i][j] / 255.0;
109      }
110    }
111    return result;
112  }

114  // Based on the formula in the publication "How to interpret the sRGB color space (specified in IEC 61966-2-1) for ICC profiles"
115  // double a = 0.055, gamma = 2.4;
116  private static double[][] convertToLinear(int[][] imageValues, double a, double gamma) {
117    double[][] result = new double[imageValues.length][imageValues[0].length];
118    for (int i = 0; i < imageValues.length; i++) {
119      for (int j = 0; j < imageValues[i].length; j++) {
120        double val = (double)imageValues[i][j] / 255.0;
121        result[i][j] = interpretSRGB(val, a, gamma);
122      }
123    }
124    return result;
125  }
126
127  private static double interpretSRGB(double value, double a, double gamma) {
128    if (value <= ALTERNATIVE_THRESHOLD) {
129      return value / 12.92;
130    } else {
131      return Math.pow((value + a) / (1 + a), gamma);
132    }
133  }

135  private static float[][] getRelativeLuminance(double[][] reds, double[][] greens, double[][] blues) {
136    float[][] result = new float[reds.length][reds[0].length];
137    for (int i = 0; i < result.length; i++) {
138      for (int j = 0; j < result[i].length; j++) {
139        result[i][j] = (float)(WEIGHT_RED * reds[i][j] + WEIGHT_GREEN * greens[i][j] + WEIGHT_BLUE * blues[i][j]);
140      }
141    }
142    return result;
143  }

145  private static int[][] relativeLuminanceToColorDomain(float[][] imageValues) {
146    int[][] result = new int[imageValues.length][imageValues[0].length];
147    for (int i = 0; i < imageValues.length; i++) {
148      for (int j = 0; j < imageValues[i].length; j++) {
149        result[i][j] = (int)(255 * imageValues[i][j]);
150      }
151    }
152    return result;
153  }
