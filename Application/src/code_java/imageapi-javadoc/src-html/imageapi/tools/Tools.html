010  /**
011   * Get the pixel matrix from a BufferedImage object
012   * @param image image
013   * @return      pixel matrix
014   */
015  public static int[][] getPixelsFast(BufferedImage image) {
016    int width = image.getWidth();
017    int[] values = image.getRGB(0, 0, width, image.getHeight(), null, 0, width);
018    return distributeRows(values, width);
019  }

022  public static int[][] distributeRows(int[] values, int width) {
023    int[][] result = new int[values.length / width][width];
024    for (int i = 0; i < result.length; i++) {
025      for (int j = 0; j < result[i].length; j++) {
026        result[i][j] = values[(i * width) + j];
027      }
028    }
029    return result;
030  }

032  public static int[][] from256toARGB(CompositeImage image) {
033    return from256toARGB(image.getMatrix());
034  }

036  // Duplicates a single color channel into every channel (ARGB-format) Sets A to 255.
037  public static int[][] from256toARGB(int[][] imageValues) {
038    int[][] result = new int[imageValues.length][imageValues[0].length];
039    for (int i = 0; i < imageValues.length; i++) {
040      for (int j = 0; j < imageValues[i].length; j++) {
041        result[i][j] = imageValues[i][j];
042        result[i][j] += (imageValues[i][j] << 8);
043        result[i][j] += (imageValues[i][j] << 16);
044        result[i][j] += (255 << 24);
045      }
046    }
047    return result;
048  }
049
050  // Generates random values for a single color channel
051  public static CompositeImage makeNoise(int width, int height) {
052    int[][] array = new int[height][width];
053    Random rng = new Random();
054
055    for (int i = 0; i < height; i++) {
056      for (int j = 0; j < width; j++) {
057        array[i][j] = rng.nextInt(256);
058      }
059    }
060    return new CompositeImage(from256toARGB(array));
061  }
062
063  // Generates random values for all color channels
064  public static CompositeImage makeColorfulNoise(int width, int height) {
065    int[][] array = new int[height][width];
066    Random rng = new Random();
067    for (int i = 0; i < height; i++) {
068      for (int j = 0; j < width; j++) {
069        array[i][j] = (255 << 24);
070        array[i][j] += (rng.nextInt(256) << 16);
071        array[i][j] += (rng.nextInt(256) << 8);
072        array[i][j] += rng.nextInt(256);
073      }
074    }
075    return new CompositeImage(array);
076  }

078  public static CompositeImage copy(CompositeImage source) {

082  public static int[][] copy(int[][] array) {
083    int[][] temp = new int[array.length][array[0].length];
084    for (int i = 0; i < array.length; i++) {
085      for (int j = 0; j < array[i].length; j++) {
086        temp[i][j] = array[i][j];
087      }
088    }
089    return temp;
090  }
091
092  // Converts from linear to ARGB
093  public static int[][] convertToARGB(double[][] array) {
094    int[][] temp = new int[array.length][array[0].length];
095    for (int i = 0; i < array.length; i++) {
096      for (int j = 0; j < array[i].length; j++) {
097        temp[i][j] = fromLinearToARGB(array[i][j]);
098      }
099    }
100    return temp;
101  }

103  // Only for a single color channel
104  public static void displayArray(int[][] array, int channel) {
105    System.out.print("\n");
106
107    for (int i = 0; i < array.length; i++) {
108      for (int j = 0; j < array[i].length; j++) {
109        if (((array[i][j] >> (channel * 8)) & 0xFF) < 10) {
110          System.out.print(" 00" + ((array[i][j] >> (channel * 8)) & 0xFF) + " ");
111        } else if (((array[i][j] >> (channel * 8)) & 0xFF) < 100) {
112          System.out.print(" 0" + ((array[i][j] >> (channel * 8)) & 0xFF) + " ");
113        } else {
114          System.out.print(" " + ((array[i][j] >> (channel * 8)) & 0xFF) + " ");
115        }
116      }
117      System.out.print("\n");
118    }
119  }

121  // Converts the greyscale linear value [0 - 1], into ARGB-integer (duplicates greyscale value to RGB color channels).
122  public static int fromLinearToARGB(double num) {
123    int val = (int)(num * 255.0);
124    return (255 << 24 | ((val & 0xFF) << 16) | ((val & 0xFF) << 8) | ((val) & 0xFF));
125  }

127  /**
128   * Maps a numerical value from one domain, with a set maximum value, to another.
129   * @param  in            value to map
130   * @param  fromDomain    current domain
131   * @param  toDomain      new domain
132   * @return               mapped value
133   */
134  public static int mapper(int in, int fromDomain, int toDomain) {
135    double ratio = ((double)(in + 1)/(double)fromDomain);
136    return (int)(ratio * (double)toDomain) - 1;
137  }
