008  // Decodes RAW16-data from Byte[] into short[]
009  public static short[][] raw16decoder (byte[] bytes, int width) {
010    int length = bytes.length / 2;
011    short[] pixels = new short[length];
012
013    for (int i = 0; i < pixels.length; i++) {
014      pixels[i] = (short)(bytes[2 * i] << 8);
015      pixels[i] += bytes[(2 * i) + 1];
016    }
017    return distributeRows(pixels, width);
018  }

021  // Decodes RAW12-data from Byte[] into short[all elements]
022  public static short[][] raw12decoder (byte[] bytes, int width) {
023    int x = 0;
024
025    Double calculationLength = new Double(bytes.length * (2.0 / 3.0));
026    int length = calculationLength.intValue();
027
028    short[] pixels = new short[length];
029
030    short pixel1, pixel2;
031    for (int i = 0; i < bytes.length; i += 3) {
032      pixel1 = (short)bytes[i];
033      pixel2 = (short)bytes[i+1];
034      pixel1 <<= 4; pixel2 <<= 4;
035
036      String lastByte = toByteString(bytes[i+2]);
037
038      short lastBytePart1 = parseBitString(lastByte.substring(0,4));
039      short lastBytePart2 = parseBitString(lastByte.substring(4,8));
040
041      pixel2 += lastBytePart1;
042      pixel1 += lastBytePart2;
043
044      pixels[x] = pixel1;
045      pixels[x+1] = pixel2;
046      x += 2;
047    }
048    return distributeRows(pixels, width);
049  }

051  // Parse a bit-String (as in "01010101" as a String object) into an integer (short).
052  public static short parseBitString(String bitString) {
053    short result = 0;
054
055    int max = bitString.length() - 1;
056    for (int i = 0; i < bitString.length(); i++) {
057      result += Math.pow(2, max-i) * (Character.getNumericValue(bitString.charAt(i)));
058    }
059    return result;
060  }
061
062  // Converts short[all elements] into short[height][width]
063  public static short[][] distributeRows(short[] list, int width) {
064    int rowNumber = -1;
065    int x = 0;
066
067    Double calculationHeight = new Double(list.length / width);
068    short height = calculationHeight.shortValue();
069
070    short[][] result = new short[height][width];
071
072    for (int i = 0; i < list.length; i++) {
073      if (i % width == 0) {
074        rowNumber++;
075      }
076      x = i - (width * rowNumber);
077      result[rowNumber][x] = list[i];
078    }
079
080    return result;
081  }

083  // Reads a String with 8-bit integer values and returns a Byte[]-matrix
084  // NOTE: Assuming that the values are seperated with spaces
085  public static byte[] parseImageData(String in) {
086    String[] values = in.split(" ");
087    byte[] byteValues = new byte[values.length];
088
089    for (int i = 0; i < values.length; i++) {
090      byteValues[i] = Byte.decode(values[i]);
091    }
092    return byteValues;
093  }

107  // Converts a byte-integer (whose leading zeros might be cut off) into a String representation, and adds leading zeros into a 8-digit number
108  public static String toByteString(byte b) {
109    Integer integer = new Integer(b);
110    String bitString = Integer.toBinaryString(integer);
111    String resultString = "00000000" + bitString;
112    resultString = resultString.substring(bitString.length());
113
114    return resultString;
115  }

117  // Converts a short-integer's bit-pattern into a String representation, and adds leading zeros into a 16-digit number
118  public static String toByteString(short s) {
119
120    Integer integer = new Integer(s);
121    String bitString = Integer.toBinaryString(integer);
122    String resultString = "0000000000000000" + bitString;
123    resultString = resultString.substring(resultString.length()-16, resultString.length());
124
125    return resultString;
126  }

128  // Quickly convert Byte[] to String[]
129  public static String[] toStringBytes(byte[] bytes) {
130    String[] result = new String[bytes.length];
131    for (int i = 0; i < result.length; i++) {
132      result[i] = toByteString(bytes[i]);
133    }
134    return result;

137  // Assumes input-string is already formatted to "XXXXXXXX" (eight digits)
138  public static String formatByteString(String in, int splitPer) {
139
140    int numSpaces = in.length()/splitPer - 1;
141    if (numSpaces < 0) {
142      return in;
143    } else {
144      String[] segment = new String[numSpaces + 1];
145      for (int i = 0; i < numSpaces + 1; i++) {
146        segment[i] = in.substring(i * splitPer, i * splitPer + splitPer);
147      }
148
149      String out = new String(segment[0]);
150      for (int j = 1; j < segment.length; j++) {
151        out += " " + segment[j];
152      }
153      return out;
154    }
155  }

158  public static String convertHyphensToNegatives(String string) {
159    return string.replace('-', '\u002D');

162  public static void printResultMatrix(short[][] matrix, boolean bin, int split) {
163    for (int i = 0; i < matrix.length; i++) {
164      for (int j = 0; j < matrix[i].length; j++) {
165        if (bin) System.out.println(formatByteString(toByteString(matrix[i][j]), split));
166        else System.out.println(matrix[i][j]);
167      }
168      System.out.println();
169    }
170  }
