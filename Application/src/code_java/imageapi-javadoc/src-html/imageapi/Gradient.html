025  /**
026   * Find the global maximum and minimum value of a single component image and normalize with regards to that value.
027   * @param data  integer pixel matrix
028   * @return      zero-to-one-normalized values
029   */
030  public static float[][] minimizeByExtremes(int[][] data) {
031    int max = 0;
032    int min = 255;
033
034    for (int i = 0; i < data.length; i++) {
035      for (int j = 0; j < data[i].length; j++) {
036        if (((data[i][j]) & 0xFF) > max) max = ((data[i][j]) & 0xFF);
037        if (((data[i][j]) & 0xFF) < min) min = ((data[i][j]) & 0xFF);
038      }
039    }
040    float fmax = (float)(max);
041    float fmin = (float)(min);
042    float fdiff = fmax - fmin;
043
044    float[][] result = new float[data.length][data[0].length];
045    for (int i = 0; i < data.length; i++) {
046      for (int j = 0; j < data[i].length; j++) {
047        result[i][j] = ((data[i][j] & 0xFF) - fmin) / fdiff;
048      }
049    }
050    return result;
051  }

053  /**
054   * Normalize values with respect to the highest single-channel value (255).
055   * <p>
056   * If an image has a narrow luminance spectrum
057   * @param  data  integer pixel data (single channel)
058   * @return       zero-to-one-normalized matrix
059   */
060  public static float[][] minimize256(int[][] data) {
061    float[][] result = new float[data.length][data[0].length];
062    for (int i = 0; i < data.length; i++) {
063      for (int j = 0; j < data[i].length; j++) {
064        result[i][j] = (float)(data[i][j]) / 255.0f;
065      }
066    }
067    return result;
068  }

091  private static int[][] gradientsToPixels(float[][] gradientMap) {
092    int[][] result = new int[gradientMap.length][gradientMap[0].length];
093    for (int i = 0; i < result.length; i ++) {
094      for (int j = 0; j < result[i].length; j++) {
095        result[i][j] = (int)(127.5f + gradientMap[i][j] * 127.5f);
096      }
097    }
098    return Tools.from256toARGB(result);
099  }

116  /**
117   * Gets the result of a gradient operation on a single-channel pixel matrix.
118   * @param  channel  integer pixel matrix (single channel)
119   * @return          gradient matrix
120   */
121  public static int[][] getGradient(int[][] channel) {
122    // float[][] minimalizedGreyscale = minimize255(greyscale);
123    int width = channel[0].length;
124    int height = channel.length;
125    float[][] minimizedChannel = minimizeByExtremes(channel);
126    float[] column = new float[height], row = new float[width];
127
128    float[][] gradientHorizontal = new float[height][width];
129    float[][] gradientVertical = new float[width][height];
130    float[][] gradientSum = new float[height][width];
131
132    System.out.println("ROW: " + row.length + "\n" + "COLUMN: " + column.length);
133
134    for (int i = 0; i < height; i++) {
135      for (int j = 0; j < width; j++) {
136        row[j] = minimizedChannel[i][j];
137      }
138      gradientHorizontal[i] = convoluteAxis(row);
139    }
140    for (int i = 0; i < width; i++) {
141      for (int j = 0; j < height; j++) {
142        column[j] = minimizedChannel[j][i];
143      }
144      gradientVertical[i] = convoluteAxis(column);
145    }
146
147    for (int i = 0; i < gradientSum.length; i++) {
148      for (int j = 0; j < gradientSum[i].length; j++) {
149        gradientSum[i][j] = (gradientVertical[j][i] + gradientHorizontal[i][j]) / 2;
150      }
151    }
152
153    return gradientsToPixels(gradientHorizontal);
154  }
