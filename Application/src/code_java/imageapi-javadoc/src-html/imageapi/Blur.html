033/**
034 * The Blur class contains methods for handling blur requests for an image or matrix. As of now there are Gaussian and linear blur methods.
035 */
036public class Blur extends ImageOperation {
037
038  /**
039   * Constant used in linear blur. All values in the traversal matrix are set to this.
040   */
041  private static final double LINEAR_BLUR_BOX_VALUE = 1.0;
042
043  /**
044   * A 7 x 7, precalculated Gaussian distribution array used as the traversal matrix in gaussian blur.
045   * <p>
046   * Retrieved from <a>https://en.wikipedia.org/wiki/Gaussian_blur</a>, 00:10 13/05/2017.
047   */
048  private static final double[][] GAUSSIAN_BLUR_BOX =
049    {{0.00000067, 0.00002292, 0.00019117, 0.00038771, 0.00019117, 0.00002292, 0.00000067},
050     {0.00002292, 0.00078634, 0.00655965, 0.01330373, 0.00655965, 0.00078633, 0.00002292},
051     {0.00019117, 0.00655965, 0.05472157, 0.11098164, 0.05472157, 0.00655965, 0.00019117},
052     {0.00038771, 0.01330373, 0.11098164, 0.22508352, 0.11098164, 0.01330373, 0.00038771},
053     {0.00019117, 0.00655965, 0.05472157, 0.11098164, 0.05472157, 0.00655965, 0.00019117},
054     {0.00002292, 0.00078633, 0.00655965, 0.01330373, 0.00655965, 0.00078633, 0.00002292},
055     {0.00000067, 0.00002292, 0.00019117, 0.00038771, 0.00019117, 0.00002292, 0.00000067}
056    };
057
058  /**
059   * The constant grid size for the gaussian blur traversal matrix.
060   */
061  private static final int GRID_SIZE_GAUSS = 7;
062
063  /**
064   * The minimum length of the traversal matrix dimensions.
065   */
066  private static final int GRID_SIZE_MINIMUM = 3;
067
068  /**
069   * The linear blur choose value.
070   * <p>
071   * Equal to 1.
072   */
073  public static final int LINEAR_BLUR = 1;
074
075  /**
076   * The gaussian blur choose value.
077   * <p>
078   * Equal to 2.
079   */
080  public static final int GAUSSIAN_BLUR = 2;
081
082  /**
083   * Compares the dimension of an image with the set grid length of the traversal matrix.
084   * @param  dimensionSize length or height of the image
085   * @param  gridSize      length of the traversal matrix dimensions
086   * @return               true if image dimension is larger
087   */
088  private static boolean sizeCheck(int dimensionSize, int gridSize) {
089    return (gridSize < dimensionSize);
090  }
091
092  /**
093   * Sets up traversal grid by blur type.
094   * <p>
095   * Types: Linear blur, Gaussian blur
096   * <p>
097   * LINEAR_BLUR and GAUSSIAN_BLUR (or their integer values) are the only valid input arguments.
098   * @param gridSize size of traversal matrix
099   * @param blurType type of blur
100   * @return         traversal matrix
101   */
102  private static double[][] setUpGrid(int gridSize, int blurType) {
103    if (blurType == LINEAR_BLUR) {
104      double[][] box = new double[gridSize][gridSize];
105
106      for (int i = 0; i < box.length; i++) {
107        for (int j = 0; j < box[i].length; j++) {
108          box[i][j] = LINEAR_BLUR_BOX_VALUE;
109        }
110      }
111      return box;
112    } else if (blurType == GAUSSIAN_BLUR) {
113      return GAUSSIAN_BLUR_BOX;
114
115    } else {
116      return null;
117    }
118  }
119
120  /**
121   * Applies a linear blur to an image with the specified grid size of the traversal matrix.
122   * @param  image         image
123   * @param  gridSize      size of traversal matrix dimensions
124   * @return               blurred image
125   */
126  public static CompositeImage linearBlur(CompositeImage image, int gridSize) {
127    return new CompositeImage(linearBlur(image.getMatrix(), gridSize));
128  }
129
130  /**
131   * Applies a linear blur to an integer pixel matrix with the specified grid size of the traversal matrix.
132   * @param  image         matrix
133   * @param  gridSize      size of traversal matrix dimensions
134   * @return               blurred image
135   */
136  public static int[][] linearBlur(int[][] image, int gridSize) {
137    // return linearBlur(new CompositeImage(image), gridSize);
138
139    int width = image[0].length;
140    int height = image.length;
141    int[][] newArray;
142
143    if (!sizeCheck(width, gridSize) || !sizeCheck(height, gridSize)) {
144      newArray = null;
145      System.err.println("gridSize is smaller than the image!");
146
147    } else {
148      newArray = new int[height][width];
149
150      double[][] box = setUpGrid(gridSize, LINEAR_BLUR);
151
152      // Same multiplier value in each "cell" of the array
153      for (int i = 0; i < gridSize; i++) {
154        for (int j = 0; j < gridSize; j++) {
155          box[i][j] = LINEAR_BLUR_BOX_VALUE;
156        }
157      }
158
159      int maxOffset = gridSize/2;
160      int[][] temp = new int[gridSize][gridSize];
161
162      // Jumps into the image a distance maxOffset, and quits the same distance
163      // This is because of unavailable values to fill the active grid which averages values inside it. Very good english. Yes.
164      // Fix this. Make a larger image?
165
166      for (int i = maxOffset; i < height - maxOffset; i++) {
167        for (int j = maxOffset; j < width - maxOffset; j++) {
168
169          for (int k = 0; k < gridSize; k++) {
170            for (int l = 0; l < gridSize; l++) {
171              temp[k][l] = image[i - maxOffset + k][j - maxOffset + l];
172            }
173          }
174          newArray[i][j] = blurOperation(temp, box);
175        }
176      }
177    }
178    return newArray;
179
180  }

182  /**
183   * Applies a gaussian blur to an image a specified number of times.
184   * <p>
185   * Uses a precalculated 7 x 7 traversal matrix.
186   * @param  image         image
187   * @param  iterations    iterations of gaussian blur
188   * @return               blurred image
189   */
190  public static CompositeImage gaussBlur(CompositeImage image, int iterations) {
191    return new CompositeImage(gaussBlur(image.getMatrix(), iterations));
192  }
193
194  /**
195   * Applies a gaussian blur to an integer pixel matrix a specified number of times.
196   * @param  image         matrix
197   * @param  iterations    iterations of gaussian blur
198   * @return               blurred matrix
199   */
200  public static int[][] gaussBlur(int[][] image, int iterations) {
201    // return gaussBlur(new CompositeImage(image), iterations);
202    int width = image[0].length;
203    int height = image.length;
204
205    if (!sizeCheck(width, GRID_SIZE_GAUSS) || !sizeCheck(height, GRID_SIZE_GAUSS)) {
206      return linearBlur(image, GRID_SIZE_MINIMUM);
207    } else {
208      int[][] newArray = new int[height][width];
209
210      // first argument into setUpGrid(int, int) is ignored.
211      double[][] box = setUpGrid(0, GAUSSIAN_BLUR);
212
213      int maxOffset = GRID_SIZE_GAUSS / 2;
214      int[][] temp = new int[GRID_SIZE_GAUSS][GRID_SIZE_GAUSS];
215
216      int[][] mediator = Tools.copy(image);
217
218      for (int count = 0; count < iterations; count++) {
219        for (int i = maxOffset; i < height - maxOffset; i++) {
220          for (int j = maxOffset; j < width - maxOffset; j++) {
221
222            for (int k = 0; k < GRID_SIZE_GAUSS; k++) {
223              for (int l = 0; l < GRID_SIZE_GAUSS; l++) {
224                temp[k][l] = mediator[i - maxOffset + k][j - maxOffset + l];
225              }
226            }
227            newArray[i][j] = blurOperation(temp, box);
228          }
229        }
230        mediator = newArray;
231      }
232      return newArray;
233    }
234  }
235
236  /**
237   * Performs an averaging operation on the image partition with the specified multiplier matrix.
238   * @param  multiplicand       partition of the original image
239   * @param  multiplier         traversal (multiplier) matrix
240   * @return                    average value
241   */
242  public static int blurOperation(int[][] multiplicand, double[][] multiplier) {
243
244    double divisor = 0;
245    for (int i = 0; i < multiplier.length; i++) {
246      for (int j = 0; j < multiplier[0].length; j++) {
247        divisor += multiplier[i][j];
248      }
249    }
250
251    double sumRed = 0, sumGreen = 0, sumBlue = 0;
252
253    CompositeImage multiplicandImage = new CompositeImage(multiplicand);
254
255    int[][] reds = multiplicandImage.getColorChannel(1);
256    int[][] greens = multiplicandImage.getColorChannel(2);
257    int[][] blues = multiplicandImage.getColorChannel(3);
258
259    for (int i = 0; i < multiplicand.length; i++) {
260      for (int j = 0; j < multiplicand[0].length; j++) {
261        sumRed += reds[i][j] * multiplier[i][j];
262        sumGreen += greens[i][j] * multiplier[i][j];
263        sumBlue += blues[i][j] * multiplier[i][j];
264      }
265    }
266
267    return ((255 << 24) + ((int)(sumRed/divisor) << 16) + ((int)(sumGreen/divisor) << 8) + (int)(sumBlue/divisor));
268  }
