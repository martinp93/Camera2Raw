002package imageapi.help;
003
004import imageapi.*;
005import imageapi.tools.*;
006
007/**
008 * This class contains documentation for terms used in the rest of the API.
009 * <p>
010 * There is no use of this in any type of image processing, but to serve as a type of javadoc index page.
011 * <p>
012 * Although there is already an index page in generated javadocs, this will be a placeholder for the terms used as not to cause confusion within other classes.
013 */
014public class Terminology {
015  /**
016   * A decomposed image where the individual values in its matrix contain information of a single color. This is often the same RAW data retrieved from an image sensor.
017   */
018  public BayerImage BAYER_IMAGE;
019
020  /**
021  * A composite image is an image where the color composites of a pixel (Alpha channel, Reds, Greens and Blues) are contained in single values (32-bit integer numbers).
022  * <p>
023  * This is opposed to a Bayer Image, where each individual value in its matrix represents only a single channel of color.
024  */
025  public CompositeImage COMPOSITE_IMAGE;
026
027  /**
028  * Used to denote a single color component in an entire image or a single pixel.
029  * <p>
030  * "Channel" and "Component" may be used interchangeably, though "channel" is used to refer more to the hexadecimal digit pair in an integer value (represented hexadeciamally).
031  * <p>
032  * "A single color channel" / "A single color component."
033  */
034  public String CHANNEL, COMPONENT;
035
036  /**
037  * The integer pixel contains a number representing the color information of a pixel.
038  * <p>
039  * The value spans from 0x00000000 to 0xFFFFFFFF (hexadecimal notation), which is 32-bits.
040  * <p>
041  * Each pair of two hexadecimal digits represents a color channel. In an example: 0xFFAABBCC, the digits FF represent a channel, the digits AA represent another and so on.
042  * <p>
043  * The color channels are: Alpha, Red, Green, Blue, or ARGB for short. And in our example over the pairs follow this abbreviation. (FF - Alpha, AA - Red, BB - Green, CC - Blue).
044  * <p>
045  * Every pair of two hexadecimal digits make up 8 bits of depth, 256 different values int the domain [0 - 255].
046  */
047  public int INTEGER_PIXEL;
048
049  /**
050  * The integer pixel matrix contains integer pixel values in a grid with the same dimensions of its image. It is essentially the image data.
051  */
052  public int[][] INTEGER_PIXEL_MATRIX;
053
054  /**
055   * Equivalent to the principle of an array or a grid; a two-dimensional object containing values with height and width; a rectangle with smaller rectangles inside it, and each rectangle holds a value; boxes with numbers within a bigger box, capiche?
056   */
057  public int[][] MATRIX;
058
059  /**
060   * To map a value from it's original domain (eg [0 - 255]), to the zero-to-one-domain.
061   * <p>
062   * Pseudo code:
063   * <p> maximum possible value = 20;
064   * <p> value = 5;
065   * <p> new_value = normalize(value);
066   * <p> reading out new_value gives us 0.25, because 5 / 20 = 0.25.
067   * <p>
068   * The normalizing operation is just a simple division, and the ratio between the value and its possible maximum is the normalized value.
069   * <p>
070   * This ratio can also be used to map the value to other domains if we multiply it with a new maximum.
071   *
072   */
073  public String NORMALIZE;
074
075  /**
076   * A pixel is referred to as the smallest component in an image, containing information for all color channels at that point.
077   * <p>
078   * If you zoom in extremely close on an image, the single-coloured squares that would be able to differentiate are these pixels.
079   * <p>
080   * Pixels are also the unit of length and size in an image. ie 200 x 200 pixels would be a square image with dimensions of length 200, and its <em>size</em> would be product of these: 200 x 200 = 40 000 pixels.
081   */
082  public String PIXEL;
083
084  /**
085  * A design pattern used to describe any class that encapsulates underlying code logic for the sake of abstraction and useability.
086  */
087  public String WRAPPER_CLASS;
088}
