020/**
021 * Test program for the image processing API.
022 */
023public class TestProgram extends JPanel {
024  /**
025   *  Has no effect on rest of program. Safely ignore this.
026   *  <p>
027   *  A non-declared serialVerionUID causes compilation warning.
028   */
029  static final long serialVersionUID = 1L;
030
031  /**
032   * Width of the canvas.
033   */
034  int width;
035
036  /**
037   * Height of the canvas.
038   */
039  int height;
040
041  /**
042   * Scale factor which deternmines the scale of each drawn pixel.
043   * <p>
044   * Uses drawRect() in Graphics2D class.
045   */
046  static int scale = 1;
047
048  /**
049   * Array of integer pixel elements.
050   */
051  static int[][] canvas;
052
053  public TestProgram() {
054    width = 0;
055    height = 0;
056  }
057
058  public TestProgram(CompositeImage composite) {
059    width = composite.getWidth();
060    height = composite.getHeight();
061
062    canvas = composite.getMatrix();
063  }
064
065  public TestProgram(BufferedImage image) {
066    width = image.getWidth();
067    height = image.getHeight();
068
069    canvas = Tools.getPixelsFast(image);
070  }
071
072  public TestProgram(int[][] composite) {
073    width = composite[0].length;
074    height = composite.length;
075    canvas = composite;
076  }
077
078  // Assuming that short[][] is 12-bit RAW format
079  public TestProgram(short[][] colorData) {
080    canvas = new int[colorData.length][colorData[0].length];
081    for (int i = 0; i < colorData.length; i++) {
082      for (int j = 0; j < colorData[i].length; j++) {
083        canvas[i][j] = Tools.mapper(colorData[i][j], 4096, 256);
084      }
085    }
086    Tools.from256toARGB(canvas);
087    // System.out.println(canvas.length + " " + canvas[0].length);
088    width = canvas[0].length;
089    height = canvas.length;
090    scale = 5;
091  }
092
093  // Generate image
094  public TestProgram(int width, int height, int scale, boolean colors) {
095    this.width = width;
096    this.height = height;
097    TestProgram.scale = scale;
098
099    if (colors) canvas = Blur.linearBlur(Tools.makeColorfulNoise(width, height), 7).getMatrix();
100    else canvas = Blur.linearBlur(Tools.makeNoise(width, height), 7).getMatrix();
101  }
102
103  public TestProgram(int average, int channel) {
104    width = 1;
105    height = 1;
106
107    scale = 50;
108    canvas = new int[1][1];
109
110    switch (channel) {
111      case 0: Tools.from256toARGB(canvas);
112              break;
113      case 1: canvas[0][0] = 255 << 24;
114              canvas[0][0] += average << 16;
115              break;
116      case 2: canvas[0][0] = 255 << 24;
117              canvas[0][0] += average << 8;
118              break;
119      case 3: canvas[0][0] = 255 << 24;
120              canvas[0][0] += average;
121              break;
122    }
123  }
124
125  public void paintComponent(Graphics g) {
126    super.paintComponent(g);
127    Graphics2D g2d = (Graphics2D) g;
128
129    for (int i = 0; i < canvas.length; i++) {
130      for (int j = 0; j < canvas[i].length; j++) {
131        Color color = new Color(canvas[i][j]);
132        g2d.setColor(color);
133        g2d.fillRect(j*scale, i*scale, scale, scale);
134        // g2d.scale();
135      }
136    }
137  }
138
139  public int getCanvasWidth() {
140    return width;
141  }
142
143  public int getCanvasHeight() {
144    return height;
145  }
146
147  public int getCanvasScale() {
148    return scale;
149  }
150
151  public void setCanvasWidth(int newWidth) {
152    width = newWidth;
153  }
154
155  public void setCanvasHeight(int newHeight) {
156    height = newHeight;
157  }
158
159  public void setCanvasScale(int newScale) {
160    scale = newScale;
161  }
162
163  // MISC. METHODS
164  public void printValues() {
165    for (int i = 0; i < height; i++) {
166      for (int j = 0; j < width; j++) {
167        System.out.println(canvas[i][j]);
168      }
169    }
170  }

173
174    JFrame frame = new JFrame("Sketch");

176    System.out.println();

180    System.out.println("   / / /  __(__  ) /_   / ____/ /  / /_/ / /_/ / /  / /_/ / / / / / /");
181    System.out.println("  /_/  \\___/____/\\__/  /_/   /_/   \\____/\\__, /_/   \\__,_/_/ /_/ /_/ ");
182    System.out.println("                                  /____/");
183
184    System.out.println();
185    System.out.println(" 1 - Generate random image");
186    System.out.println(" 2 - Load image");
187    System.out.println(" 3 - Test Algebraic Color To Greyscale");
188    System.out.println(" 4 - Test gauss blur iterations");
189    System.out.println();
190
191
192    do {
193
194      Console c = System.console();
195      int option = Integer.parseInt(c.readLine("Choose one: "));
196
197      TestProgram test;
198      test = null;
199
200      if (option == 1) {
201        // Generate random image
202        int width, height, scale;
203        boolean color;
204        System.out.println("Grey\tColor");
205        System.out.println("1\t2");
206
207        int choice = Integer.parseInt(c.readLine("Setting: "));
208        if (choice == 1) color = false;
209        else if (choice == 2) color = true;
210        else break;
211
212        width = Integer.parseInt(c.readLine("Width of Array: "));
213        height = Integer.parseInt(c.readLine("Height of Array: "));
214        scale = Integer.parseInt(c.readLine("Scale factor: "));
215
216        test = new TestProgram(width, height, scale, color);
217
218      } else if (option == 2) {
219        BufferedImage image = ImageIO.read(new File("media/testimage_small.jpg"));
220        test = new TestProgram(image);
221      } else if (option == 3) {
222        CompositeImage composite = new CompositeImage(ImageIO.read(new File("media/testimage_smaller.jpg")));
223
224        ALGEBRAIC_COLOR_TO_GREYSCALE grey = new ALGEBRAIC_COLOR_TO_GREYSCALE(composite);
225        composite = grey.getResult();
226        test = new TestProgram(composite);
227
228      } else if (option == 4) {
229        CompositeImage composite = new CompositeImage(ImageIO.read(new File("media/testimage_even_smaller.jpg")));
230        composite = Blur.gaussBlur(composite, 5);
231        // test.setCanvasScale(2);
232        test = new TestProgram(composite);
233      } else {
234        System.exit(0);
235      }
236
237      test.setLocation(500, 500);
238
239      frame.add(test);
240      // frame.repaint(0, 0, 0, test.gtX(), test.gtY());
241      frame.setExtendedState(JFrame.MAXIMIZED_BOTH);
242      // frame.setSize(test.width * test.scale, test.height * test.scale);
243      frame.setVisible(true);
244
245    } while (true);
246  }
